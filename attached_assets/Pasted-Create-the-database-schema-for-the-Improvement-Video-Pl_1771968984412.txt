Create the database schema for the Improvement Video Platform — the Paul Akers-inspired 60-second video sharing system.

CREATE A NEW MIGRATION with these tables:

TABLE: improvement_videos
- id: uuid, primary key, default gen_random_uuid()
- store_id: text, not null, references stores(id)
- employee_id: text, not null (Clerk user ID)
- title: text, not null (max 200 chars)
- description: text, nullable (max 1000 chars)
- category: text, not null (enum-like: 'process', 'workspace', 'customer_experience', 'visual_merchandising', 'inventory', 'equipment', 'other')
- storage_type: text, not null (enum-like: 'youtube', 's3')
- youtube_video_id: text, nullable (YouTube video ID if storage_type = 'youtube')
- s3_key: text, nullable (S3 object key if storage_type = 's3')
- s3_url: text, nullable (CloudFront URL if storage_type = 's3')
- thumbnail_url: text, nullable
- duration_seconds: integer, nullable
- status: text, not null, default 'processing' (enum-like: 'processing', 'ready', 'failed')
- is_featured: boolean, default false (AI-picked "Improvement of the Week")
- view_count: integer, default 0
- created_at: timestamp with timezone, default now()

TABLE: video_likes
- id: uuid, primary key, default gen_random_uuid()
- video_id: uuid, not null, references improvement_videos(id) ON DELETE CASCADE
- employee_id: text, not null (Clerk user ID)
- created_at: timestamp with timezone, default now()
- UNIQUE(video_id, employee_id) — one like per person per video

TABLE: video_comments
- id: uuid, primary key, default gen_random_uuid()
- video_id: uuid, not null, references improvement_videos(id) ON DELETE CASCADE
- employee_id: text, not null (Clerk user ID)
- comment_text: text, not null (max 500 chars)
- created_at: timestamp with timezone, default now()

INDEXES:
- improvement_videos(store_id, created_at DESC)
- improvement_videos(store_id, status, category)
- improvement_videos(employee_id, created_at DESC)
- improvement_videos(store_id, is_featured)
- video_likes(video_id)
- video_likes(employee_id)
- video_comments(video_id, created_at)

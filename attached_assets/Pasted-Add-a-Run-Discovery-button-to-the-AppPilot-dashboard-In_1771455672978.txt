Add a "Run Discovery" button to the AppPilot dashboard
In the AppPilot admin dashboard, add a button that lets the owner manually trigger a codebase discovery cycle from the browser.
1. Add a new API endpoint in node_modules/apppilot/src/middleware/admin-dashboard.js:
javascriptapp.post(`${basePath}/api/discover`, authMiddleware, async (req, res) => {
  // Run a heartbeat with forceDiscovery: true
  const { runHeartbeatCycle } = require('../orchestrator.js');
  
  res.json({ status: 'started', message: 'Discovery cycle started...' });
  
  // Run in background so the request doesn't timeout
  runHeartbeatCycle({
    apiKey: process.env.ANTHROPIC_API_KEY,
    appUrl: `http://localhost:${process.env.PORT || 5000}`,
    forceDiscovery: true,
    projectRoot: projectRoot,
  }).then(result => {
    if (memoryStore) {
      memoryStore.logDaily(`Manual discovery triggered from dashboard: ${result.status}`);
    }
  }).catch(err => {
    if (memoryStore) {
      memoryStore.logDaily(`Manual discovery failed: ${err.message}`);
    }
  });
});
Register this route in the mount function alongside the other routes.
2. Add a button to the dashboard HTML in node_modules/apppilot/src/dashboard/index.html:
Add a prominent "Run Discovery" button near the top of the dashboard, next to the status info. When clicked:

Button text changes to "Running..." with a spinner
Button becomes disabled
Makes a POST to /apppilot/api/discover
After response, show a success message: "Discovery started — refresh in 30s to see results"
Re-enable button after 60 seconds (cooldown to prevent spam)

Also add a "Run Continuous" button next to it that triggers the continuous loop:
javascriptapp.post(`${basePath}/api/continuous`, authMiddleware, async (req, res) => {
  const { runContinuousLoop } = require('../orchestrator.js');
  
  res.json({ status: 'started', message: 'Continuous improvement loop started...' });
  
  runContinuousLoop({
    apiKey: process.env.ANTHROPIC_API_KEY,
    appUrl: `http://localhost:${process.env.PORT || 5000}`,
    maxCycles: 5,
    budgetMax: 2.0,
    projectRoot: projectRoot,
  }).then(result => {
    if (memoryStore) {
      memoryStore.logDaily(`Continuous loop completed: ${result.totalCycles} cycles, ${result.totalChangesApplied} changes`);
    }
  }).catch(err => {
    if (memoryStore) {
      memoryStore.logDaily(`Continuous loop failed: ${err.message}`);
    }
  });
});

# Add ELON.md — Strategic Constraint Solver for AppPilot (with Playwright)

## What ELON.md Is

ELON.md is AppPilot’s strategic brain. Instead of finding random bugs and fixing them, ELON reads GOALS.md and SOUL.md, figures out the fastest path to reaching those goals, identifies what’s actually blocking progress (the limiting factor), and relentlessly focuses the agent on solving that ONE constraint before moving to the next.

It works like Elon Musk’s “idiot index” — find the bottleneck, remove it, find the next bottleneck, repeat. No context switching, no scattered fixes. Laser focus on whatever is most blocking the goal.

**ELON also uses Playwright to crawl the live site** — it doesn’t just read code, it actually uses the app like a real user to find what’s broken and verify that fixes actually work from the user’s perspective.

## The Flow

```
ELON reads GOALS.md + SOUL.md + codebase
    ↓
Crawls the live site with Playwright to find real user-facing issues
    ↓
Combines code analysis + crawl results to identify the #1 limiting factor
    ↓
Breaks the limiting factor into concrete tasks
    ↓
Feeds tasks as specs to Ralph Loop
    ↓
Ralph Loop solves them one by one
    ↓
ELON re-crawls with Playwright to verify: is the constraint actually fixed for users?
    ↓
If NO → keep working on it
    ↓
If YES → identify the NEXT limiting factor → repeat
    ↓
Stop when all goals for current phase are achieved or budget is exhausted
```

```javascript
'use strict';

const { chromium } = require('playwright');

async function crawlSite(config = {}) {
  const {
    appUrl = 'http://localhost:5000',
    maxPages = 30,
    timeout = 10000,
    waitForNetworkIdle = true,
  } = config;

  const browser = await chromium.launch({ headless: true });
  const context = await browser.newContext({
    viewport: { width: 375, height: 812 }, // iPhone size — mobile first
    userAgent: 'AppPilot-ELON/1.0 Playwright Crawler',
  });

  const page = await context.newPage();

  const results = {
    pagesVisited: 0,
    errors: [],
    warnings: [],
    performance: [],
    brokenLinks: [],
    consoleErrors: [],
    networkFailures: [],
    screenshots: {},
  };

  const visited = new Set();
  const toVisit = [appUrl];

  // Capture console errors
  page.on('console', msg => {
    if (msg.type() === 'error') {
      results.consoleErrors.push({
        type: 'console-error',
        message: msg.text(),
        url: page.url(),
        timestamp: new Date().toISOString(),
        severity: 'high',
      });
    }
  });

  // Capture uncaught exceptions
  page.on('pageerror', err => {
    results.errors.push({
      type: 'uncaught-exception',
      message: err.message,
      stack: err.stack,
      url: page.url(),
      timestamp: new Date().toISOString(),
      severity: 'high',
    });
  });

  // Capture failed network requests
  page.on('requestfailed', request => {
    results.networkFailures.push({
      type: 'network-failure',
      message: `${request.method()} ${request.url()} failed: ${request.failure()?.errorText}`,
      url: page.url(),
      requestUrl: request.url(),
      method: request.method(),
      timestamp: new Date().toISOString(),
      severity: 'high',
    });
  });

  // Capture HTTP errors from API calls
  page.on('response', response => {
    if (response.status() >= 400) {
      const url = response.url();
      // Only track same-origin API calls
      if (url.startsWith(appUrl)) {
        results.errors.push({
          type: 'api-error',
          message: `${response.request().method()} ${url} returned ${response.status()}`,
          url: page.url(),
          requestUrl: url,
          statusCode: response.status(),
          timestamp: new Date().toISOString(),
          severity: response.status() >= 500 ? 'high' : 'medium',
        });
      }
    }
  });

  while (toVisit.length > 0 && results.pagesVisited < maxPages) {
    const url = toVisit.shift();
    if (visited.has(url)) continue;
    visited.add(url);

    try {
      const startTime = Date.now();

      const response = await page.goto(url, {
        timeout,
        waitUntil: waitForNetworkIdle ? 'networkidle' : 'domcontentloaded',
      });

      const loadTime = Date.now() - startTime;
      results.pagesVisited++;

      // Check page load time
      if (loadTime > 3000) {
        results.performance.push({
          type: 'slow-page',
          message: `Page took ${loadTime}ms to load`,
          url,
          loadTime,
          timestamp: new Date().toISOString(),
          severity: loadTime > 5000 ? 'high' : 'medium',
        });
      }

      // Check for HTTP error on the page itself
      if (response && response.status() >= 400) {
        results.errors.push({
          type: 'page-error',
          message: `Page returned ${response.status()}`,
          url,
          statusCode: response.status(),
          timestamp: new Date().toISOString(),
          severity: 'high',
        });
      }

      // Check for broken images
      const brokenImages = await page.evaluate(() => {
        const imgs = Array.from(document.querySelectorAll('img'));
        return imgs
          .filter(img => img.complete && img.naturalWidth === 0 && img.src)
          .map(img => img.src);
      });

      for (const imgSrc of brokenImages) {
        results.errors.push({
          type: 'broken-image',
          message: `Broken image: ${imgSrc}`,
          url,
          imageSrc: imgSrc,
          timestamp: new Date().toISOString(),
          severity: 'medium',
        });
      }

      // Check for empty/error states in the UI
      const uiIssues = await page.evaluate(() => {
        const issues = [];
        // Check for error messages visible to users
        const errorElements = document.querySelectorAll('[class*="error"], [class*="Error"], [role="alert"]');
        for (const el of errorElements) {
          if (el.textContent.trim()) {
            issues.push({ type: 'visible-error', text: el.textContent.trim().substring(0, 200) });
          }
        }
        // Check for empty states that shouldn't be empty
        const emptyLists = document.querySelectorAll('[class*="empty"], [class*="no-results"], [class*="no-data"]');
        for (const el of emptyLists) {
          if (el.offsetParent !== null) { // visible
            issues.push({ type: 'empty-state', text: el.textContent.trim().substring(0, 200) });
          }
        }
        return issues;
      });

      for (const issue of uiIssues) {
        results.warnings.push({
          type: issue.type,
          message: issue.text,
          url,
          timestamp: new Date().toISOString(),
          severity: 'low',
        });
      }

      // Find internal links to crawl next
      const links = await page.evaluate((baseUrl) => {
        const anchors = Array.from(document.querySelectorAll('a[href]'));
        return anchors
          .map(a => a.href)
          .filter(href => href.startsWith(baseUrl) && !href.includes('#') && !href.includes('mailto:'));
      }, appUrl);

      for (const link of links) {
        if (!visited.has(link) && !toVisit.includes(link)) {
          toVisit.push(link);
        }
      }

    } catch (err) {
      results.errors.push({
        type: 'navigation-error',
        message: `Failed to load page: ${err.message}`,
        url,
        timestamp: new Date().toISOString(),
        severity: 'high',
      });
    }
  }

  await browser.close();

  // Combine all issues into a flat list sorted by severity
  results.allIssues = [
    ...results.errors,
    ...results.consoleErrors,
    ...results.networkFailures,
    ...results.performance,
    ...results.warnings,
  ].sort((a, b) => {
    const sev = { high: 0, medium: 1, low: 2 };
    return (sev[a.severity] || 2) - (sev[b.severity] || 2);
  });

  return results;
}

// Targeted crawl — test specific pages/flows related to a constraint
async function verifyCrawl(config = {}) {
  const {
    appUrl = 'http://localhost:5000',
    pagesToCheck = [],
    checks = [],
    timeout = 10000,
  } = config;

  const browser = await chromium.launch({ headless: true });
  const context = await browser.newContext({
    viewport: { width: 375, height: 812 },
  });
  const page = await context.newPage();

  const results = {
    passed: [],
    failed: [],
  };

  for (const pageUrl of pagesToCheck) {
    const fullUrl = pageUrl.startsWith('http') ? pageUrl : `${appUrl}${pageUrl}`;

    try {
      const startTime = Date.now();
      const response = await page.goto(fullUrl, { timeout, waitUntil: 'networkidle' });
      const loadTime = Date.now() - startTime;

      // Basic checks
      if (response.status() < 400) {
        results.passed.push({ url: fullUrl, check: 'page-loads', loadTime });
      } else {
        results.failed.push({ url: fullUrl, check: 'page-loads', reason: `Status ${response.status()}` });
      }

      // Run custom checks
      for (const check of checks) {
        try {
          const checkResult = await page.evaluate(check.evaluate);
          if (checkResult) {
            results.passed.push({ url: fullUrl, check: check.name });
          } else {
            results.failed.push({ url: fullUrl, check: check.name, reason: check.failReason || 'Check returned false' });
          }
        } catch (err) {
          results.failed.push({ url: fullUrl, check: check.name, reason: err.message });
        }
      }

    } catch (err) {
      results.failed.push({ url: fullUrl, check: 'navigation', reason: err.message });
    }
  }

  await browser.close();

  return {
    allPassed: results.failed.length === 0,
    passed: results.passed.length,
    failed: results.failed.length,
    details: results,
  };
}

module.exports = { crawlSite, verifyCrawl };
```

### 3. Create the ELON subagent template

Create `node_modules/apppilot/templates/subagents/elon.md`:

```markdown
---
name: elon
description: Strategic constraint solver. Reads goals, codebase, and live site crawl data to identify the single biggest limiting factor.
model: sonnet
---

You are ELON — AppPilot's strategic constraint solver. Your job is NOT to find bugs or improvements. Your job is to answer ONE question:

**"What is the single biggest thing blocking this app from reaching its next goal?"**

## Your Inputs

You receive:
1. **GOALS.md** — the app's mission, priorities, and current focus areas
2. **SOUL.md** — the app's identity and purpose
3. **Codebase** — actual source code from key files
4. **Crawl results** — real errors and issues found by Playwright crawling the live site as a user
5. **Previous constraints** — what's already been solved (don't repeat these)

## Your Process

1. Read GOALS.md — especially "Current Priorities" and "Focus areas this month"
2. Read the crawl results — these are REAL problems users experience right now
3. Read the codebase — understand what's built and what's missing
4. Cross-reference: which code issues cause the crawl errors? Which goals are blocked?
5. Identify the #1 LIMITING FACTOR

## What Makes a Limiting Factor

A limiting factor is the thing that BLOCKS other things from happening. Use crawl data to find what's actually broken for users:

- Crawl found 500 errors on /api/products → products can't load → blocks feed, search, checkout → HIGH constraint
- Crawl found broken images on feed → users see broken experience → blocks engagement → HIGH constraint
- Crawl found 4s page load on /feed → violates quality targets → blocks user retention → HIGH constraint
- Code has dead exports → blocks nothing → NOT a constraint

**Prioritize crawl-discovered issues** — they represent real user pain.

Ask: "If I fix this ONE thing, how many other things become possible?"

## Output Format

Respond with ONLY a JSON object. No markdown, no explanation:

{
  "currentGoal": "The specific goal from GOALS.md this analysis targets",
  "limitingFactor": {
    "description": "Clear description of what's blocking progress",
    "why": "Why this is the #1 constraint — what it blocks",
    "unblocks": ["List of things that become possible once this is solved"],
    "constraintScore": 9,
    "category": "infrastructure|security|feature|performance|integration",
    "evidenceFromCrawl": ["List of specific crawl errors that prove this is a real problem"]
  },
  "plan": [
    {
      "step": 1,
      "filePath": "server/routes.ts",
      "description": "What to change and why",
      "successCriteria": ["How to verify this step is done"],
      "priority": "high",
      "estimatedComplexity": "low|medium|high"
    }
  ],
  "verificationPages": ["/feed", "/products", "/cart"],
  "completionCriteria": "How to know the limiting factor is fully removed",
  "previousConstraints": ["List constraints already solved"]
}

### Field Requirements:
- **constraintScore**: 1-10, where 10 = blocks everything, 1 = blocks almost nothing
- **plan**: Ordered list of concrete steps (max 10). Each becomes a spec for Ralph Loop.
- **verificationPages**: Pages to crawl after fixing to verify the constraint is removed
- **evidenceFromCrawl**: Specific errors from the crawl that prove this matters
- **completionCriteria**: Testable condition proving the constraint is removed

### Rules:
- ALWAYS pick the constraint with the HIGHEST score (most blocking)
- If two constraints tie, pick the one with the most crawl evidence
- Each step must be a single-file edit — Ralph Loop handles one file at a time
- Never suggest steps requiring manual work — everything must be automatable
- Skip constraints in "Ignore for now" section of GOALS.md
- Prioritize issues found by the Playwright crawl — they affect real users
```

### 4. Create the ELON engine

Create `node_modules/apppilot/src/elon.js`:

This is the core engine. Key integration points with Playwright:

```javascript
'use strict';

const fs = require('fs');
const path = require('path');
const { delegateToSubagent } = require('./subagents/dispatcher');
const { loadContext } = require('./context-loader');
const { crawlSite, verifyCrawl } = require('./site-crawler');

async function runElonCycle(config) {
  const {
    apiKey,
    appUrl,
    projectRoot = process.cwd(),
    dataDir = path.join(projectRoot, '.apppilot'),
    maxSpecs = 10,
    budgetMax = 5.0,
    enableCrawl = true,
  } = config;

  const context = loadContext(projectRoot);
  const elonLog = _loadElonLog(dataDir);
  const budget = { spent: 0, max: budgetMax };

  // Step 1: Crawl the live site with Playwright
  let crawlResults = null;
  if (enableCrawl) {
    console.log('🕷️  Crawling site with Playwright...');
    try {
      crawlResults = await crawlSite({ appUrl, maxPages: 30, timeout: 10000 });
      console.log(`   Crawled ${crawlResults.pagesVisited} pages, found ${crawlResults.allIssues.length} issues`);
      // Save crawl results
      fs.mkdirSync(dataDir, { recursive: true });
      fs.writeFileSync(
        path.join(dataDir, 'last-crawl.json'),
        JSON.stringify(crawlResults, null, 2)
      );
    } catch (err) {
      console.log(`   Crawl failed: ${err.message} — continuing with code analysis only`);
    }
  }

  // Step 2: Read key source files
  const sourceCode = _readSourceFiles(projectRoot);

  // Step 3: Ask ELON to identify the limiting factor
  const task = {
    goals: context.goals ? context.goals.content : 'No GOALS.md found',
    soul: context.soul ? context.soul.content : 'No SOUL.md found',
    codebase: sourceCode,
    crawlResults: crawlResults ? {
      pagesVisited: crawlResults.pagesVisited,
      totalIssues: crawlResults.allIssues.length,
      highSeverity: crawlResults.allIssues.filter(i => i.severity === 'high'),
      mediumSeverity: crawlResults.allIssues.filter(i => i.severity === 'medium'),
      // Truncate to avoid token limits
      topIssues: crawlResults.allIssues.slice(0, 20).map(i => ({
        type: i.type,
        message: i.message,
        url: i.url,
        severity: i.severity,
      })),
    } : { note: 'Crawl not available — analyze code only' },
    previousConstraints: elonLog.solved || [],
    currentConstraint: elonLog.current || null,
  };

  let analysis;
  try {
    analysis = await delegateToSubagent('elon', task, {
      context,
      budget,
      apiKey,
      identityDir: projectRoot,
      templatesDir: path.join(projectRoot, 'node_modules', 'apppilot', 'templates'),
    });
  } catch (err) {
    return { status: 'failed', reason: 'ELON analysis failed: ' + err.message };
  }

  if (!analysis || !analysis.limitingFactor) {
    return { status: 'failed', reason: 'ELON could not identify a limiting factor' };
  }

  // Step 4: Log the constraint
  const constraint = {
    id: 'constraint-' + Date.now(),
    description: analysis.limitingFactor.description,
    why: analysis.limitingFactor.why,
    unblocks: analysis.limitingFactor.unblocks,
    score: analysis.limitingFactor.constraintScore,
    category: analysis.limitingFactor.category,
    evidenceFromCrawl: analysis.limitingFactor.evidenceFromCrawl || [],
    goal: analysis.currentGoal,
    identifiedAt: new Date().toISOString(),
    status: 'active',
    steps: analysis.plan || [],
    verificationPages: analysis.verificationPages || [],
    completionCriteria: analysis.completionCriteria,
  };

  // Step 5: Convert plan steps into specs
  const specsCreated = [];
  const approvedDir = path.join(dataDir, 'approved-queue');
  const pendingDir = path.join(dataDir, 'queue', 'pending');
  fs.mkdirSync(approvedDir, { recursive: true });
  fs.mkdirSync(pendingDir, { recursive: true });

  for (const step of analysis.plan) {
    if (specsCreated.length >= maxSpecs) break;

    const spec = {
      filePath: step.filePath,
      description: '[ELON] ' + step.description,
      successCriteria: step.successCriteria,
      testCommand: 'curl -s http://localhost:' + (process.env.PORT || 5000) + '/health',
      elonConstraintId: constraint.id,
      priority: step.priority,
      step: step.step,
    };

    const needsApproval = _needsOwnerApproval(step, context);
    const targetDir = needsApproval ? pendingDir : approvedDir;
    const prefix = needsApproval ? 'pending' : 'approved';

    const filename = 'elon-' + constraint.id + '-step' + String(step.step).padStart(2, '0') + '.json';
    fs.writeFileSync(path.join(targetDir, filename), JSON.stringify(spec, null, 2));
    specsCreated.push({ filename, dir: prefix, description: step.description, needsApproval });
  }

  // Step 6: Update logs and ELON.md
  elonLog.current = constraint;
  if (!elonLog.history) elonLog.history = [];
  elonLog.history.push(constraint);
  _saveElonLog(dataDir, elonLog);
  _writeElonMd(projectRoot, constraint, elonLog);

  return {
    status: 'planned',
    constraint: {
      description: constraint.description,
      score: constraint.score,
      unblocks: constraint.unblocks,
      evidenceFromCrawl: constraint.evidenceFromCrawl,
    },
    crawl: crawlResults ? {
      pagesVisited: crawlResults.pagesVisited,
      issuesFound: crawlResults.allIssues.length,
    } : null,
    specsCreated: specsCreated.length,
    specsAutoApproved: specsCreated.filter(s => !s.needsApproval).length,
    specsPendingApproval: specsCreated.filter(s => s.needsApproval).length,
    plan: specsCreated,
  };
}

// After fixes are applied, use Playwright to VERIFY the constraint is actually resolved
async function evaluateConstraint(config) {
  const {
    apiKey,
    appUrl,
    projectRoot = process.cwd(),
    dataDir = path.join(projectRoot, '.apppilot'),
    enableCrawl = true,
  } = config;

  const context = loadContext(projectRoot);
  const elonLog = _loadElonLog(dataDir);

  if (!elonLog.current) {
    return { status: 'no-active-constraint' };
  }

  const constraint = elonLog.current;

  // Check completed/failed specs
  const completedDir = path.join(dataDir, 'completed');
  const failedDir = path.join(dataDir, 'failed');

  let completedSteps = 0;
  let failedSteps = 0;

  if (fs.existsSync(completedDir)) {
    completedSteps = fs.readdirSync(completedDir).filter(f => f.includes(constraint.id)).length;
  }
  if (fs.existsSync(failedDir)) {
    failedSteps = fs.readdirSync(failedDir).filter(f => f.includes(constraint.id)).length;
  }

  const totalSteps = constraint.steps.length;
  const remainingSteps = totalSteps - completedSteps - failedSteps;

  // If all steps done, use Playwright to verify the fix actually works
  if (remainingSteps <= 0) {
    let crawlVerification = null;

    if (enableCrawl && constraint.verificationPages && constraint.verificationPages.length > 0) {
      console.log('🔍 Verifying fix with Playwright...');
      try {
        crawlVerification = await verifyCrawl({
          appUrl,
          pagesToCheck: constraint.verificationPages,
        });
        console.log('   Verification: ' + crawlVerification.passed + ' passed, ' + crawlVerification.failed + ' failed');
      } catch (err) {
        console.log('   Verification crawl failed: ' + err.message);
      }
    }

    // Also do a full crawl to see if the specific errors are gone
    let reCrawl = null;
    if (enableCrawl) {
      console.log('🕷️  Re-crawling site to confirm fix...');
      try {
        reCrawl = await crawlSite({ appUrl, maxPages: 15, timeout: 10000 });
        console.log('   Re-crawl: ' + reCrawl.allIssues.length + ' issues (was: check previous crawl)');
      } catch (err) {
        console.log('   Re-crawl failed: ' + err.message);
      }
    }

    // Ask ELON to evaluate based on code + crawl results
    const sourceCode = _readSourceFiles(projectRoot);
    const verifyTask = {
      constraint: constraint,
      completionCriteria: constraint.completionCriteria,
      codebase: sourceCode,
      crawlVerification: crawlVerification,
      reCrawlResults: reCrawl ? {
        pagesVisited: reCrawl.pagesVisited,
        totalIssues: reCrawl.allIssues.length,
        highSeverity: reCrawl.allIssues.filter(i => i.severity === 'high').length,
        issues: reCrawl.allIssues.slice(0, 10),
      } : null,
      question: 'Has the limiting factor been fully resolved? Check the completion criteria against the current code AND the crawl results. Are the original crawl errors gone?',
    };

    try {
      const result = await delegateToSubagent('elon', verifyTask, {
        context,
        budget: { spent: 0, max: 1.0 },
        apiKey,
        identityDir: projectRoot,
        templatesDir: path.join(projectRoot, 'node_modules', 'apppilot', 'templates'),
      });

      const resolved = result && (
        result.resolved === true ||
        (result.raw && result.raw.toLowerCase().includes('resolved'))
      );

      if (resolved) {
        constraint.status = 'solved';
        constraint.solvedAt = new Date().toISOString();
        constraint.verificationResult = {
          crawlPassed: crawlVerification ? crawlVerification.allPassed : null,
          issuesAfterFix: reCrawl ? reCrawl.allIssues.length : null,
        };
        if (!elonLog.solved) elonLog.solved = [];
        elonLog.solved.push(constraint.description);
        elonLog.current = null;
        _saveElonLog(dataDir, elonLog);
        _writeElonMd(projectRoot, null, elonLog);

        return {
          status: 'constraint-resolved',
          constraint: constraint.description,
          verification: constraint.verificationResult,
        };
      }
    } catch (err) {
      // Verification failed, keep working
    }

    // Not fully resolved — check if crawl still shows related errors
    return {
      status: 'partially-resolved',
      constraint: constraint.description,
      completedSteps,
      failedSteps,
      remainingSteps: 0,
      totalSteps,
      crawlIssuesRemaining: reCrawl ? reCrawl.allIssues.filter(i => i.severity === 'high').length : null,
      needsMoreWork: true,
    };
  }

  return {
    status: 'in-progress',
    constraint: constraint.description,
    completedSteps,
    failedSteps,
    remainingSteps,
    totalSteps,
  };
}

// Full autonomous ELON loop
async function runElonLoop(config) {
  const {
    apiKey,
    appUrl,
    projectRoot = process.cwd(),
    dataDir = path.join(projectRoot, '.apppilot'),
    maxConstraints = 5,
    budgetMax = 10.0,
    cooldownSeconds = 30,
    enableCrawl = true,
  } = config;

  const results = {
    constraintsSolved: 0,
    totalChangesApplied: 0,
    totalBudgetUsed: 0,
    constraints: [],
    stoppedReason: null,
  };

  const { runHeartbeatCycle } = require('./orchestrator');

  for (let c = 0; c < maxConstraints; c++) {
    console.log('\n=== ELON Constraint ' + (c + 1) + '/' + maxConstraints + ' (budget: $' + results.totalBudgetUsed.toFixed(2) + '/$' + budgetMax + ') ===');

    if (results.totalBudgetUsed >= budgetMax) {
      results.stoppedReason = 'budget-exhausted';
      break;
    }

    // Step 1: Identify limiting factor (with crawl)
    console.log('Identifying limiting factor...');
    const elonResult = await runElonCycle({
      apiKey,
      appUrl,
      projectRoot,
      dataDir,
      budgetMax: budgetMax - results.totalBudgetUsed,
      enableCrawl,
    });

    if (elonResult.status === 'failed') {
      console.log('ELON analysis failed: ' + elonResult.reason);
      results.stoppedReason = 'analysis-failed';
      break;
    }

    console.log('Limiting factor: "' + elonResult.constraint.description + '" (score: ' + elonResult.constraint.score + '/10)');
    console.log('Unblocks: ' + elonResult.constraint.unblocks.join(', '));
    if (elonResult.crawl) {
      console.log('Crawl evidence: ' + elonResult.crawl.issuesFound + ' issues found across ' + elonResult.crawl.pagesVisited + ' pages');
    }
    console.log('Created ' + elonResult.specsCreated + ' specs (' + elonResult.specsAutoApproved + ' auto-approved, ' + elonResult.specsPendingApproval + ' pending)');

    if (elonResult.specsPendingApproval > 0 && elonResult.specsAutoApproved === 0) {
      console.log('All specs need owner approval. Stopping.');
      results.stoppedReason = 'approval-needed';
      results.constraints.push(elonResult);
      break;
    }

    // Step 2: Run heartbeats to process specs
    let maxHeartbeats = 10;
    let changesThisConstraint = 0;

    for (let h = 0; h < maxHeartbeats; h++) {
      if (results.totalBudgetUsed >= budgetMax) break;

      console.log('  Running heartbeat ' + (h + 1) + '/' + maxHeartbeats + '...');

      const hbResult = await runHeartbeatCycle({ apiKey, appUrl, projectRoot });
      results.totalBudgetUsed += hbResult.budgetUsed || 0;

      const queueStep = hbResult.steps.find(s => s.step === 'approved_queue');
      const processed = queueStep && queueStep.result ? queueStep.result.processed : 0;
      changesThisConstraint += processed;

      if (processed === 0) {
        console.log('  No more specs to process.');
        break;
      }

      console.log('  Applied ' + processed + ' changes.');

      if (h < maxHeartbeats - 1) {
        console.log('  Cooling down ' + cooldownSeconds + 's...');
        await new Promise(r => setTimeout(r, cooldownSeconds * 1000));
      }
    }

    results.totalChangesApplied += changesThisConstraint;

    // Step 3: Use Playwright to verify the constraint is resolved
    console.log('Verifying constraint with Playwright...');
    const evalResult = await evaluateConstraint({
      apiKey,
      appUrl,
      projectRoot,
      dataDir,
      enableCrawl,
    });

    if (evalResult.status === 'constraint-resolved') {
      console.log('✅ Constraint resolved: "' + elonResult.constraint.description + '"');
      if (evalResult.verification) {
        console.log('   Crawl verification: ' + (evalResult.verification.crawlPassed ? 'PASSED' : 'ISSUES REMAIN'));
      }
      results.constraintsSolved++;
    } else if (evalResult.status === 'partially-resolved') {
      console.log('⚠️  Partially resolved — crawl still shows ' + (evalResult.crawlIssuesRemaining || '?') + ' high-severity issues');
      // TODO: generate more specs to fully resolve
    } else {
      console.log('⏳ In progress (' + evalResult.completedSteps + '/' + evalResult.totalSteps + ' steps)');
    }

    results.constraints.push({
      ...elonResult,
      changesApplied: changesThisConstraint,
      resolved: evalResult.status === 'constraint-resolved',
      verification: evalResult,
    });

    if (c < maxConstraints - 1) {
      console.log('\nCooling down ' + cooldownSeconds + 's before next constraint...');
      await new Promise(r => setTimeout(r, cooldownSeconds * 1000));
    }
  }

  if (!results.stoppedReason) {
    results.stoppedReason = results.constraintsSolved >= maxConstraints
      ? 'max-constraints-reached'
      : 'all-constraints-resolved';
  }

  console.log('\n=== ELON Complete ===');
  console.log('Constraints solved: ' + results.constraintsSolved);
  console.log('Total changes: ' + results.totalChangesApplied);
  console.log('Budget used: $' + results.totalBudgetUsed.toFixed(2));
  console.log('Stopped: ' + results.stoppedReason);

  return results;
}

function _readSourceFiles(projectRoot) {
  const extensions = ['.ts', '.tsx', '.js', '.jsx'];
  const dirs = ['server', 'src', 'shared', 'client/src', 'lib', 'api'];
  let content = '';
  const maxTotal = 50000;

  for (const dir of dirs) {
    const fullDir = path.join(projectRoot, dir);
    if (!fs.existsSync(fullDir)) continue;
    try {
      const files = fs.readdirSync(fullDir).filter(f => extensions.some(e => f.endsWith(e)));
      for (const file of files) {
        if (content.length >= maxTotal) break;
        try {
          const fc = fs.readFileSync(path.join(fullDir, file), 'utf8');
          content += '\n// === ' + dir + '/' + file + ' ===\n' + fc.substring(0, 8000) + '\n';
        } catch {}
      }
    } catch {}
  }
  return content.substring(0, maxTotal);
}

function _loadElonLog(dataDir) {
  try {
    return JSON.parse(fs.readFileSync(path.join(dataDir, 'elon-log.json'), 'utf8'));
  } catch {
    return { current: null, solved: [], history: [] };
  }
}

function _saveElonLog(dataDir, log) {
  fs.mkdirSync(dataDir, { recursive: true });
  fs.writeFileSync(path.join(dataDir, 'elon-log.json'), JSON.stringify(log, null, 2));
}

function _writeElonMd(projectRoot, currentConstraint, log) {
  let md = '# ELON.md — Current Constraint Focus\n\n';
  md += '<!-- Auto-generated by AppPilot ELON engine. Do not edit. -->\n\n';

  if (currentConstraint) {
    md += '## 🎯 Active Constraint\n\n';
    md += '**' + currentConstraint.description + '**\n\n';
    md += '- Score: ' + currentConstraint.score + '/10\n';
    md += '- Category: ' + currentConstraint.category + '\n';
    md += '- Goal: ' + currentConstraint.goal + '\n';
    md += '- Why: ' + currentConstraint.why + '\n\n';

    if (currentConstraint.evidenceFromCrawl && currentConstraint.evidenceFromCrawl.length > 0) {
      md += '### Evidence (from Playwright crawl)\n';
      for (const e of currentConstraint.evidenceFromCrawl) {
        md += '- ' + e + '\n';
      }
      md += '\n';
    }

    md += '### Unblocks\n';
    for (const u of (currentConstraint.unblocks || [])) {
      md += '- ' + u + '\n';
    }
    md += '\n### Plan\n';
    for (const s of (currentConstraint.steps || [])) {
      md += s.step + '. ' + s.description + ' (' + s.filePath + ')\n';
    }

    if (currentConstraint.verificationPages) {
      md += '\n### Verification Pages\n';
      for (const p of currentConstraint.verificationPages) {
        md += '- ' + p + '\n';
      }
    }

    md += '\n### Completion Criteria\n' + currentConstraint.completionCriteria + '\n';
  } else {
    md += '## ✅ No Active Constraint\n\nAll identified constraints have been resolved.\n';
  }

  if (log.solved && log.solved.length > 0) {
    md += '\n## Solved Constraints\n';
    for (const s of log.solved) {
      md += '- ✅ ' + s + '\n';
    }
  }

  fs.writeFileSync(path.join(projectRoot, 'ELON.md'), md);
}

function _needsOwnerApproval(step, context) {
  if (!context.goals || !context.goals.content) return true;
  const goals = context.goals.content.toLowerCase();
  const approvalSection = goals.split('always require my approval')[1];
  if (!approvalSection) return false;
  const approvalBlock = approvalSection.split('###')[0].toLowerCase();
  const keywords = ['auth', 'payment', 'checkout', 'schema', 'new api', 'oauth', 'webhook', 'ai prompt', 'graphql', 'billing', 'provisioning', 'role', 'permission'];
  const desc = step.description.toLowerCase();
  const file = (step.filePath || '').toLowerCase();
  return keywords.some(k => desc.includes(k) || file.includes(k));
}
```

```javascript
    }
```
